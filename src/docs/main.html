 40def get_coords(count, y_pos, x_min, x_max):
 41    """Gera coordenadas (x, y) uniformemente distribu√≠das numa faixa horizontal.
 42
 43    Esta fun√ß√£o n√£o est√° atualmente a ser usada no layout principal, mas
 44    representa uma forma gen√©rica de distribuir agentes na horizontal.
 45
 46    Args:
 47        count (int): N√∫mero de pontos a gerar.
 48        y_pos (float): Coordenada Y fixa para todos os pontos.
 49        x_min (float): Limite m√≠nimo de X.
 50        x_max (float): Limite m√°ximo de X.
 51
 52    Returns:
 53        list[tuple[float, float]]: Lista de pares (x, y) espa√ßados
 54        uniformemente entre `x_min` e `x_max`. Se `count == 0`, devolve lista
 55        vazia.
 56    """
 57    if count == 0:
 58        return []
 59    x_values = np.linspace(x_min, x_max, count + 2)[1:-1]
 60    return [(round(x, 2), y_pos) for x in x_values]

150        name = f"M{i+1}"

152            f"machine{i+1}@{domain}", pwd, env=env, name=name,
153            position=pos, suppliers=suppliers_jids,

155            maintenance_crews=maintenance_jids, robots=robots_jids
156        )
157        machines.append(m)
158        machines_jids.append(str(m.jid))
159
160    all_agents = machines + robots + suppliers + maintenance_crews
161
162    return all_agents, machines_jids

379    """
