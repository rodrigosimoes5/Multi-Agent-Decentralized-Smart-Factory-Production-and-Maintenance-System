 13
 14    Este agente representa uma equipa de manutenção que responde a pedidos
 15    de reparação de máquinas. Utiliza o protocolo ``maintenance-cnp``:

 82    class MaintenanceParticipant(CyclicBehaviour):
 83        """Comportamento CNP para responder a pedidos de manutenção.
 84
 85        Este comportamento trata da parte de negociação:
 86
 87        - Recebe mensagens com o protocolo ``"maintenance-cnp"``.
 88        - Para mensagens com performative ``"cfp"``:
 89            * Se a equipa estiver disponível → calcula um custo e faz PROPOSE.
 90            * Caso contrário → responde REFUSE, indicando o estado atual.
 91        - Para mensagens com performative ``"accept-proposal"``:
 92            * Regista a reparação atual e define o número de ticks restantes.
 93            * Atualiza o estado da equipa para ``"busy"``.
 94            * Opcionalmente envia um INFORM com ``"repair_started"``.
 95        - Para mensagens com performative ``"reject-proposal"``:
 96            * Apenas regista a rejeição no log.
 97        """
 98
 99        async def run(self):
100            """Processa uma mensagem do protocolo de manutenção.
101
102            Este método é chamado ciclicamente pelo motor de comportamentos
103            do SPADE. Em cada iteração:
104
105            1. Tenta ler uma mensagem com timeout curto.
106            2. Se não houver mensagem ou se não for do protocolo
107               ``"maintenance-cnp"``, termina a iteração.
108            3. Com base no performative (``cfp``, ``accept-proposal``,
109               ``reject-proposal``), executa a lógica correspondente descrita
110               em :class:`MaintenanceParticipant`.
111
112            Protocolo de mensagens:
113                - CFP:
114                    * metadata:
115                        - ``protocol`` = ``"maintenance-cnp"``
116                        - ``performative`` = ``"cfp"``
117                        - ``thread`` = ID do leilão/negociação
118                    * body (JSON):
119                        - ``machine_jid`` (str)
120                        - ``machine_pos`` (list[float, float])
121                - PROPOSE:
122                    * metadata:
123                        - ``protocol`` = ``"maintenance-cnp"``
124                        - ``performative`` = ``"propose"``
125                        - ``thread`` = mesmo ID da CFP
126                    * body (JSON):
127                        - ``crew`` (str): nome da equipa.
128                        - ``cost`` (float): custo estimado.
129                        - ``repair_time`` (int): duração estimada da reparação.
130                        - ``distance`` (float): distância à máquina.
131                - REFUSE:
132                    * metadata:
133                        - ``protocol`` = ``"maintenance-cnp"``
134                        - ``performative`` = ``"refuse"``
135                        - ``thread`` = mesmo ID da CFP
136                    * body (JSON, opcional):
137                        - ``reason`` (str): motivo da recusa (por ex., ``"busy"``).
138                - ACCEPT-PROPOSAL / REJECT-PROPOSAL:
139                    * metadata:
140                        - ``protocol`` = ``"maintenance-cnp"``
141                        - ``performative`` = ``"accept-proposal"`` ou
142                          ``"reject-proposal"``
143                        - ``thread`` = mesmo ID da CFP
144                    * body:
145                        - No caso de ``accept-proposal``: inclui informação
146                          da máquina e ``repair_time``.
147            """
148            agent = self.agent
149            msg = await self.receive(timeout=0.5)
150            if not msg:
151                return
152
153            if msg.metadata.get("protocol") != "maintenance-cnp":
154                return
155
156            pf = msg.metadata.get("performative")
157            thread_id = msg.metadata.get("thread")
158
159            try:
160                data = json.loads(msg.body) if msg.body else {}
161            except Exception:
162                data = {}
163
164            if pf == "cfp":
165                machine_jid = data.get("machine_jid")
166                machine_pos = data.get("machine_pos", [0, 0])
167
168                if agent.crew_status != "available":
169                    rep = Message(to=str(msg.sender))
170                    rep.set_metadata("protocol", "maintenance-cnp")
171                    rep.set_metadata("performative", "refuse")
172                    rep.set_metadata("thread", thread_id)
173                    rep.body = json.dumps({"reason": agent.crew_status})
174                    await self.send(rep)
175
176                    await agent.log(
177                        f"[MAINT-CNP] REFUSE para {machine_jid} (status={agent.crew_status})"
178                    )
179                    return
180
181                dx = machine_pos[0] - agent.position[0]
182                dy = machine_pos[1] - agent.position[1]
183                distance = math.sqrt(dx * dx + dy * dy)
184
185                repair_time = random_int_in_range(agent.repair_time_range)
186                cost = distance + repair_time
187
188                rep = Message(to=str(msg.sender))
189                rep.set_metadata("protocol", "maintenance-cnp")
190                rep.set_metadata("performative", "propose")
191                rep.set_metadata("thread", thread_id)
192                rep.body = json.dumps(
193                    {
194                        "crew": agent.agent_name,
195                        "cost": cost,
196                        "repair_time": repair_time,
197                        "distance": distance,
198                    }
199                )
200                await self.send(rep)
201
202                await agent.log(
203                    f"[MAINT-CNP] PROPOSE para {machine_jid}: "
204                    f"cost={cost:.2f}, repair_time={repair_time}, dist={distance:.2f}"
205                )
206                return
207
208            if pf == "accept-proposal":
209                machine_jid = data.get("machine_jid")
210                machine_pos = data.get("machine_pos", [0, 0])
211                repair_time = int(data.get("repair_time", 5))
212
213                agent.current_repair = {
214                    "machine_jid": machine_jid,
215                    "machine_pos": machine_pos,
216                    "thread": thread_id,
217                }
218                agent.repair_ticks_remaining = repair_time
219                agent.crew_status = "busy"
220
221                await agent.log(
222                    f"[MAINT-CNP] ACCEPT de {machine_jid}. "
223                    f"Iniciar reparação ({repair_time} ticks)."
224                )
225
226                inf = Message(to=machine_jid)
227                inf.set_metadata("protocol", "maintenance-cnp")
228                inf.set_metadata("performative", "inform")
229                inf.set_metadata("thread", thread_id)
230                inf.body = json.dumps({"status": "repair_started"})
231                await self.send(inf)
232                return
233
234            # REJECT-PROPOSAL → apenas logar
235            if pf == "reject-proposal":
236                await agent.log("[MAINT-CNP] REJECT recebido.")
237                return

239    class RepairExecutor(CyclicBehaviour):
240        """Comportamento responsável por simular a execução da reparação.
241
242        Este comportamento:
243
244        - Verifica se existe uma reparação em curso (`current_repair`).
245        - Se não houver, marca a equipa como ``"available"`` e fica em espera.
246        - Se houver, decrementa `repair_ticks_remaining` a cada execução,
247          simulando a passagem de um tick de reparação.
248        - Quando o número de ticks chegar a zero:
249            * Envia uma mensagem INFORM com ``"repair_completed"`` para a
250              máquina correspondente.
251            * Atualiza o estado da equipa para ``"available"``.
252            * Incrementa a métrica global ``repairs_finished`` no ambiente.
253        """
254
255        async def run(self):
256            """Atualiza o estado da reparação em curso.
257
258            Este método é chamado ciclicamente, representando o avanço de
259            um tick de simulação para a reparação:
260
261            - Se `current_repair` for ``None``:
262                * Marca a equipa como disponível.
263                * Aguarda um curto intervalo antes da próxima iteração.
264            - Se `current_repair` não for ``None``:
265                * Decrementa `repair_ticks_remaining`.
266                * Regista no log o número de ticks restantes.
267                * Quando `repair_ticks_remaining` chega a 0 ou menos:
268                    - Envia INFORM para a máquina com
269                      ``status="repair_completed"``.
270                    - Limpa `current_repair` e volta a marcar a equipa como
271                      disponível.
272                    - Atualiza a métrica `repairs_finished` no ambiente, se
273                      este existir.
274            """
275            agent = self.agent
276
277            if agent.current_repair is None:
278                agent.crew_status = "available"
279                await asyncio.sleep(0.5)
280                return
281
282            if agent.repair_ticks_remaining > 0:
283                agent.repair_ticks_remaining -= 1
284                await agent.log(
285                    f"[REPAIR] Reparando {agent.current_repair['machine_jid']}: "
286                    f"{agent.repair_ticks_remaining} ticks restantes."
287                )
288
289            if agent.repair_ticks_remaining <= 0:
290                machine_jid = agent.current_repair["machine_jid"]
291                thread_id = agent.current_repair["thread"]
292
293                inf = Message(to=machine_jid)
294                inf.set_metadata("protocol", "maintenance-cnp")
295                inf.set_metadata("performative", "inform")
296                inf.set_metadata("thread", thread_id)
297                inf.body = json.dumps({"status": "repair_completed"})
298                await self.send(inf)
299
300                await agent.log(
301                    f"[REPAIR] Reparação concluída para {machine_jid}. "
302                    "INFORM 'repair_completed' enviado."
303                )
304
305                agent.current_repair = None
306                agent.crew_status = "available"
307
308                if agent.env:
309                    agent.env.metrics["repairs_finished"] += 1
310
311            await asyncio.sleep(1)

314def random_int_in_range(rng):
315    """Gera um inteiro aleatório dentro de um intervalo.
316
317    Função auxiliar simples que devolve um número inteiro aleatório dentro
318    do intervalo dado por `rng`, usando ``random.randint``.
319
320    Args:
321        rng (tuple[int, int]): Intervalo (mín, máx) no qual gerar o número
322            inteiro aleatório.
323
324    Returns:
325        int: Número inteiro aleatório no intervalo `[rng[0], rng[1]]`.
326    """
327    import random
328    return random.randint(rng[0], rng[1])
