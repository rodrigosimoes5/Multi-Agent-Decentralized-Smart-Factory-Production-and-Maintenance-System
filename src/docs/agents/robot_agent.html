 10def euclidean_distance(p1, p2):
 11    """Calcula a distância euclidiana entre dois pontos 2D.
 12
 13    Args:
 14        p1 (Sequence[float]): Ponto inicial, como tuplo ou lista ``(x1, y1)``.
 15        p2 (Sequence[float]): Ponto final, como tuplo ou lista ``(x2, y2)``.
 16
 17    Returns:
 18        float: Distância euclidiana entre `p1` e `p2`.
 19    """
 20    dx = p2[0] - p1[0]
 21    dy = p2[1] - p1[1]
 22    return math.sqrt(dx * dx + dy * dy)

 25class RobotAgent(FactoryAgent):
 26    """Robot de transporte com gestão de energia, responsável por transportar:
 27
 28    - Materiais (dos suppliers para as máquinas).
 29    - Jobs (entre máquinas).
 30
 31    O robot participa no protocolo ``"transport-cnp"``:
 32
 33    A coordenação é feita via um protocolo CNP único, ``"transport-cnp"``,
 34em que o robot:
 35
 36    1. Recebe CFP com a descrição do transporte (tipo, origem, destino, etc.).
 37    2. Se tiver energia suficiente e estiver livre, responde com PROPOSE,
 38    indicando custo, distância e energia necessária.
 39    3. Caso contrário, responde com REFUSE (por exemplo, ``"busy"``
 40    ou ``"low_energy"``).
 41    4. Quando recebe ACCEPT-PROPOSAL, executa o transporte:
 42    - Atualiza energia.
 43    - Simula tempo de viagem.
 44    - Atualiza a sua posição.
 45    - Envia INFORM com ``status="delivered"`` ao iniciador.
 46    5. Para jobs, pode ainda notificar a máquina de destino com o próprio job
 47    via protocolo ``"job-transfer"``.
 48
 49    Inclui ainda um modelo simples de energia com recarga automática
 50    quando o robot está ocioso.
 51
 52    """

116    class TransportManagerBehaviour(CyclicBehaviour):
117        """Comportamento principal do robot para gestão de transportes.
118
119        Responsabilidades:
120
121       
122        - Receber mensagens do protocolo ``"transport-cnp"``.
123        - Para CFP:
124            * Verificar se o robot está livre e se tem energia suficiente.
125            * Calcular distância e energia necessária.
126            * Responder com PROPOSE ou REFUSE.
127        - Para ACCEPT-PROPOSAL:
128            * Executar o transporte (simulação de viagem).
129            * Atualizar energia e posição.
130            * Enviar INFORM com ``status="delivered"``.
131            * No caso de jobs, notificar também a máquina de destino
132              via protocolo ``"job-transfer"``.
133        - Para REJECT-PROPOSAL:
134            * Apenas regista a rejeição no log.
135        """

137        async def run(self):
138            """Executa um ciclo de gestão de energia e processamento de mensagens.
139
140            Em cada iteração:
141
142            
143            1. Tenta receber uma mensagem com pequeno timeout.
144            2. Se a mensagem pertencer ao protocolo ``"transport-cnp"``,
145               processa-a de acordo com o seu `performative`:
146                - ``"cfp"`` → devolve PROPOSE ou REFUSE.
147                - ``"accept-proposal"`` → executa o transporte.
148                - ``"reject-proposal"`` → apenas log.
149            """
150            agent = self.agent
151
152
153            msg = await self.receive(timeout=0.3)
154            if not msg:
155                await asyncio.sleep(0.2)
156                return
157
158            if msg.metadata.get("protocol") != "transport-cnp":
159                return
160
161            pf = msg.metadata.get("performative")
162            thread_id = msg.metadata.get("thread")
163
164            try:
165                task = json.loads(msg.body) if msg.body else {}
166            except Exception:
167                task = {}
168
169            kind = task.get("kind", "materials")
170
171            if kind == "materials":
172                origin = task.get("supplier_pos", [0, 0])
173                dest = task.get("machine_pos", [0, 0])
174            else:
175                origin = task.get("origin_pos", [0, 0])
176                dest = task.get("dest_pos", [0, 0])
177
178            d1 = euclidean_distance(agent.position, origin)
179            d2 = euclidean_distance(origin, dest)
180            total_dist = d1 + d2
181            energy_needed = max(
182                1, int(math.ceil(total_dist * agent.energy_per_unit))
183            )
184
185            if pf == "cfp":
186                if agent.busy:
187                    rep = Message(to=str(msg.sender))
188                    rep.set_metadata("protocol", "transport-cnp")
189                    rep.set_metadata("performative", "refuse")
190                    rep.set_metadata("thread", thread_id)
191                    rep.body = json.dumps({"reason": "busy"})
192                    await self.send(rep)
193                    return
194
195                if agent.energy < energy_needed or agent.energy <= agent.low_energy_threshold:
196                    rep = Message(to=str(msg.sender))
197                    rep.set_metadata("protocol", "transport-cnp")
198                    rep.set_metadata("performative", "refuse")
199                    rep.set_metadata("thread", thread_id)
200                    rep.body = json.dumps({"reason": "low_energy"})
201                    await self.send(rep)
202
203                    await agent.log(
204                        f"[ROBOT] REFUSE (low_energy) para {msg.sender} "
205                        f"(energy={agent.energy}%, needed={energy_needed}%)"
206                    )
207                    return
208
209                total_qty = 0
210                if kind == "materials":
211                    total_qty = sum(task.get("materials", {}).values())
212
213                cost = total_dist + 0.1 * total_qty
214
215                rep = Message(to=str(msg.sender))
216                rep.set_metadata("protocol", "transport-cnp")
217                rep.set_metadata("performative", "propose")
218                rep.set_metadata("thread", thread_id)
219                rep.body = json.dumps(
220                    {
221                        "cost": cost,
222                        "distance": total_dist,
223                        "energy_needed": energy_needed,
224                    }
225                )
226                await self.send(rep)
227
228                await agent.log(
229                    f"[ROBOT] PROPOSE para {msg.sender}: cost={cost:.2f}, "
230                    f"dist={total_dist:.2f}, energy_needed={energy_needed} "
231                    f"(energy atual={agent.energy}%)"
232                )
233                return
234
235            if pf == "accept-proposal":
236                agent.busy = True
237
238                if agent.energy < energy_needed:
239                    await agent.log(
240                        f"[ROBOT] ERRO: energy insuficiente na hora do ACCEPT "
241                        f"(energy={agent.energy}%, needed={energy_needed}%)."
242                    )
243                    rep = Message(to=str(msg.sender))
244                    rep.set_metadata("protocol", "transport-cnp")
245                    rep.set_metadata("performative", "refuse")
246                    rep.set_metadata("thread", thread_id)
247                    rep.body = json.dumps({"reason": "late_low_energy"})
248                    await self.send(rep)
249                    agent.busy = False
250                    return
251
252                prev_e = agent.energy
253                agent.energy = max(0, agent.energy - energy_needed)
254
255                await agent.log(
256                    f"[ROBOT] ACCEPT-PROPOSAL recebido. "
257                    f"Vai transportar kind={kind}, dist={total_dist:.2f}, "
258                    f"energy {prev_e}% → {agent.energy}%"
259                )
260
261                travel_time = max(0.5, total_dist * agent.speed)
262                await asyncio.sleep(travel_time)
263
264                agent.position = tuple(dest)
265
266                inf = Message(to=str(msg.sender))
267                inf.set_metadata("protocol", "transport-cnp")
268                inf.set_metadata("performative", "inform")
269                inf.set_metadata("thread", thread_id)
270                inf.body = json.dumps({"status": "delivered", "kind": kind})
271                await self.send(inf)
272
273                if kind == "job":
274                    job = task.get("job", {})
275                    dest_jid = task.get("destination_jid")
276
277                    if dest_jid and job:
278                        job_msg = Message(to=dest_jid)
279                        job_msg.set_metadata("protocol", "job-transfer")
280                        job_msg.set_metadata("performative", "inform")
281                        job_msg.set_metadata("thread", thread_id)
282                        job_msg.body = json.dumps(
283                            {"status": "job_delivered", "job": job}
284                        )
285                        await self.send(job_msg)
286
287                        await agent.log(
288                            f"[ROBOT] Job {job.get('id')} entregue a {dest_jid} "
289                            f"(via job-transfer)."
290                        )
291
292                await agent.log(
293                    f"[ROBOT] Transporte concluído. "
294                    f"thread={thread_id}, kind={kind}, pos_final={agent.position}"
295                )
296
297                agent.busy = False
298                return
299
300            if pf == "reject-proposal":
301                await agent.log(
302                    f"[ROBOT] REJECT recebido de {msg.sender} "
303                    f"(thread={thread_id})."
304                )
305                return
