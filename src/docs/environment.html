  5class FactoryEnvironment:
  6    """Ambiente partilhado da fábrica.
  7
  8    Esta classe representa o contexto global da simulação:
  9
 10    - `time`: contador discreto de ticks de simulação.
 11    - `metrics`: dicionário com contadores globais de desempenho.
 12    - `agents`: lista de instâncias dos agentes registados no ambiente.
 13
 14    As métricas atualmente consideradas são:
 15
 16    - Jobs / produção:
 17        * ``jobs_created``: número total de jobs criados pelo supervisor.
 18        * ``jobs_completed``: número de jobs concluídos com sucesso.
 19        * ``jobs_lost``: jobs cancelados ou perdidos.
 20    - Supply:
 21        * ``supply_requests_ok``: pedidos de materiais concluídos com sucesso.
 22        * ``supply_requests_failed``: pedidos de materiais que falharam/expiraram.
 23    - Falhas e manutenção:
 24        * ``machine_failures``: número total de falhas de máquinas.
 25        * ``repairs_started``: número de reparações iniciadas.
 26        * ``repairs_finished``: número de reparações concluídas.
 27        * ``downtime_ticks``: total de ticks em que máquinas estiveram falhadas.
 28    - Transferência de jobs:
 29        * ``jobs_transferred``: número de jobs transferidos entre máquinas
 30          com auxílio de robots.
 31
 32    """

 34    def __init__(self):
 35        """Inicializa o ambiente com tempo zero e métricas a 0."""
 36        self.time = 0
 37        self.metrics = {
 38            
 39            "jobs_created": 0,
 40            "jobs_completed": 0,
 41            "jobs_lost": 0,
 42
 43            
 44            "supply_requests_ok": 0,
 45            "supply_requests_failed": 0,
 46    
 47        
 48            "machine_failures": 0,
 49            "repairs_started": 0,
 50            "repairs_finished": 0,
 51            "downtime_ticks": 0,
 52
 53        
 54            "jobs_transferred": 0,
 55        }
 56        self.agents = []
 57
 58    def register_agent(self, agent):
 59        """Regista um agente no ambiente.
 60
 61        Normalmente chamado a partir do método `setup()` de cada agente
 62        (via :class:`FactoryAgent`), de forma a que o ambiente possa:
 63
 64        - Descobrir todos os agentes existentes.
 65        - Agregar estado (por exemplo, máquinas disponíveis).
 66        - Produzir um snapshot global para monitorização.
 67
 68        Args:
 69            agent: Instância de um agente SPADE a registar neste ambiente.
 70        """
 71        self.agents.append(agent)

 73    def get_all_agents_status(self):
 74        """Recolhe o estado atual de todos os agentes registados.
 75
 76        Produz uma estrutura de dados pronta a ser serializada em JSON,
 77        com a seguinte forma geral:
 78
 79        .. code-block:: json
 80
 81            {
 82              "metrics": { ... },
 83              "time": <int>,
 84              "agents": {
 85                "machines": [ ... ],
 86                "robots": [ ... ],
 87                "suppliers": [ ... ],
 88                "maintenance_crews": [ ... ],
 89                "supervisor": { ... }
 90              }
 91            }
 92
 93        Para cada tipo de agente, são recolhidos campos relevantes:
 94
 95        - **MachineAgent**:
 96            * name, jid, position
 97            * is_failed
 98            * queue_size, max_queue
 99            * current_job_id, current_step, ticks_remaining
100        - **RobotAgent**:
101            * name, jid, position
102            * energy, max_energy, is_busy
103        - **SupplierAgent**:
104            * name, jid, position
105            * stock, recharging
106        - **MaintenanceAgent**:
107            * name, jid, position
108            * status (available/busy)
109            * repair_target, ticks_remaining
110        - **SupervisorAgent**:
111            * jid
112            * last_job_id
113
114        Returns:
115            dict: Estrutura de estado global da simulação, adequada para
116            serialização com ``json.dumps`` ou para envio a um dashboard.
117        """
118        data = {
119            "metrics": self.metrics,
120            "time": self.time,
121            "agents": {
122                "machines": [],
123                "robots": [],
124                "suppliers": [],
125                "maintenance_crews": [],
126                "supervisor": {}
127            }
128        }
129
130        for agent in self.agents:
131            agent_type = agent.__class__.__name__
132            
133            if agent_type == "MachineAgent":
134                current_job = getattr(agent, 'current_job', None)
135                
136                if current_job:
137                    job_id = current_job.get('id')
138                    job_step_idx = current_job.get('current_step_idx', 0)
139                    pipeline = getattr(agent, 'pipeline_template', [])
140                    current_step_name = pipeline[job_step_idx] if 0 <= job_step_idx < len(pipeline) else "UNKNOWN_STEP"
141                    ticks_remaining = getattr(agent, 'current_job_ticks', 0)
142                else:
143                    job_id = None
144                    current_step_name = None
145                    ticks_remaining = 0
146                
147                state = {
148                    "name": getattr(agent, 'agent_name', None),
149                    "jid": str(agent.jid),
150                    "position": getattr(agent, 'position', (0,0)),
151                    "is_failed": getattr(agent, 'is_failed', False),
152                    "queue_size": len(getattr(agent, 'job_queue', [])),
153                    "max_queue": getattr(agent, 'max_queue', 15),
154                    "current_job_id": job_id,
155                    "current_step": current_step_name,
156                    "ticks_remaining": ticks_remaining
157                }
158                data["agents"]["machines"].append(state)
159            
160            elif agent_type == "RobotAgent":
161                state = {
162                    "name": getattr(agent, 'agent_name', None),
163                    "jid": str(agent.jid),
164                    "position": getattr(agent, 'position', (0,0)),
165                    "energy": getattr(agent, 'energy', 0),
166                    "max_energy": getattr(agent, 'max_energy', 100),
167                    "is_busy": getattr(agent, 'busy', False)
168                }
169                data["agents"]["robots"].append(state)
170                
171            elif agent_type == "SupplierAgent":
172                state = {
173                    "name": getattr(agent, 'agent_name', None),
174                    "jid": str(agent.jid),
175                    "position": getattr(agent, 'position', (0,0)),
176                    "stock": getattr(agent, 'stock', {}),
177                    "recharging": getattr(agent, 'recharging', False),
178                }
179                data["agents"]["suppliers"].append(state)
180            
181            
182            elif agent_type == "MaintenanceAgent":
183                current_repair = getattr(agent, 'current_repair', None)
184                
185                state = {
186                    "name": getattr(agent, 'agent_name', None),
187                    "jid": str(agent.jid),
188                    "position": getattr(agent, 'position', (0,0)),
189                    "status": getattr(agent, 'crew_status', 'available'),
190                    "repair_target": current_repair.get('machine_jid') if current_repair else None,
191                    "ticks_remaining": getattr(agent, 'repair_ticks_remaining', 0) if current_repair else 0
192                }
193                data["agents"]["maintenance_crews"].append(state)
194            
195            elif agent_type == "SupervisorAgent":
196                data["agents"]["supervisor"] = {
197                    "jid": str(agent.jid),
198                    "last_job_id": getattr(agent, 'job_counter', 0),
199                }
200
201        return data

203    async def tick(self):
204        """Avança o tempo global de simulação uma unidade.
205
206        Este método:
207
208        - Incrementa o contador `time` em 1 tick.
209        - Aguarda um pequeno intervalo (0.1 s) para simular o avanço do
210          tempo físico/real na simulação.
211
212        Pode ser chamado num loop principal do cenário, por exemplo:
213
214        .. code-block:: python
215
216            env = FactoryEnvironment()
217            while True:
218                await env.tick()
219        """
220        self.time += 1
221        await asyncio.sleep(0.1)
